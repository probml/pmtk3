function error = NRMSE(Xfull, Ximputed, miss)
idx = find(miss == 1);
numer = (Ximputed(idx) - Xfull(idx)).^2;
denom = (Xfull(idx)).^2;
error = sqrt(sum(numer)) / sqrt(sum(denom));
end
